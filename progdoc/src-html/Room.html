001import java.util.HashMap ;
002import java.util.Set;
003import java.util.Iterator;
004
005/**
006 * Classe Room : Salle du jeu avec sa description, ses sorties, les items qu'elle possède.
007 * 
008 * @author Wallerand 
009 * @version 11/04/2016
010 */
011public class Room
012{
013    private String aDescription;
014    private HashMap<String, Room> exits;
015    private String aImageName;
016    private ItemList aItemList;
017     
018    /**
019         * Constructeur
020         * @param pDescription
021         *      Description de la salle
022         * @param pImage Image de la salle
023         * Renvoi une salle nouvellement créée sans sorties
024         */
025    public Room(final String pDescription, String pImage){
026        this.aDescription = pDescription;
027        this.exits = new HashMap<String, Room>();
028        this.aItemList = new ItemList();
029        this.aImageName = pImage;
030    }//room natural
031    
032    
033    /**
034         * getDescription() : Renvoi la description de la salle
035         * @return description de la salle
036         */    
037    public String getDescription(){
038        return this.aDescription;
039    }//getDescription
040
041    /**
042         * setExits() :
043         * @param pDirection
044         *      Direction à ajouter
045         * @param pNeighbor
046         *      Nom de la salle voisine à ajouter
047         */    
048    public void setExits(final String pDirection, final Room pNeighbor){
049        exits.put(pDirection, pNeighbor);
050            
051    }//setExits
052
053    /**
054         * getExits
055         * @param pDirection
056         *      Direction à interroger
057         * @return Renvoi le nom de la salle qui correspond à la direction
058         */    
059    public Room getExits(final String pDirection){
060        
061        return exits.get(pDirection);
062    }//getExits
063    
064    /**
065         * getAllExits
066         * @return Renvoi la HashMap des exits
067         */    
068    public HashMap<String, Room> getAllExits(){
069  
070        return this.exits;
071    }//getExits
072    
073    /**
074         * getExitsString ()
075         * @return Renvoi les directions possibles formatées
076         */    
077    public String getExitsString(){
078        String result = "Exits: ";
079        Set<String> keys = exits.keySet();
080        for(String exit : keys){
081            result = result+exit+" ";
082        }
083        return result;
084    }
085    
086    public ItemList getItemList(){
087        return this.aItemList;
088    }
089    
090    public void setItem(final String pName, final Item pItem){
091        this.aItemList.setItem(pName, pItem);
092    }
093    
094    /**
095     * getLongDescription()
096     * @return Renvoi la description de la Room + sorties + items présents
097     */
098    public String getLongDescription(){
099        return "You are in the "+aDescription+"\n"+getExitsString()+"\nItems : "+this.aItemList.getItemsString();
100    }//getLongDescription
101    
102    /**
103     * getImageName()
104     * @return : Renvoi la valeur de aImageName
105     */
106        public String getImageName()
107        {
108                return this.aImageName;
109        }
110} // Room
