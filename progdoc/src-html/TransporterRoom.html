002/**
003 * Classe qui décrit la TransporterRoom
004 *
005 * @author Wallerand
006 * @version 23/05/2016
007 */
008public class TransporterRoom extends Room
009{
010    /**
011     * Constructeur de la classe
012     */
013    public TransporterRoom(final String pDescription, final String pImage)
014    {
015        super(pDescription, pImage);
016    }
017    /**
018     * getExits Overrited
019     * @param pDirection
020     *          Direction à interroger
021     * @return Room Renvoi la Room où aller
022     */
023    @Override
024    public Room getExits(final String pDirection){
025        return findRandomRoom();
026    }
027
028    /**
029     * findRandomRoom
030     * @return Room Renvoi une Room au hasard
031     */
032    private Room findRandomRoom(){
033        return RoomRandomizer.getRandomRoom(super.getAllExits());
034    }
035
036    /**
037     * getLongDescription()
038     * @return Renvoi la description de la Room + sorties + items présents
039     */
040    @Override
041    public String getLongDescription(){
042        return "You are in the "+super.getDescription()+"\nExits : Somewhere\nItems : "+super.getItemList().getItemsString();
043    }//getLongDescription
044}
