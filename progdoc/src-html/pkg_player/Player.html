014/**
015 * Classe du joueur. Elle contient sa pièce courante, les pièces précédentes, les items portés, le poids maximum transportable.
016 *
017 * @author Wallerand
018 * @version 24/05/2016
019 */
020public class Player
021{
022    private Room aCurrentRoom;
023    private Stack<Room> aLastRoom;
024    private int aMaxWeight;
025    private ItemList aItemList;
026    private int aTimeLimit;
027    private ItemList aMovingCharacterList;
028
029    /**
030     * Constructeur : On initialise seulement la salle courante.
031     * @param pCurrentRoom Salle d'où démarre le joueur
032     */
033    public Player(final Room pCurrentRoom)
034    {
035        this.aCurrentRoom = pCurrentRoom;
036        this.aLastRoom = new Stack<Room>();
037        this.aItemList = new ItemList();
038        this.aMovingCharacterList = new ItemList();
039        this.aMaxWeight = 4;
040        this.aTimeLimit = 20;
041    }//Player
042
043    /**
044     * getCurrentRoom() :
045     * @return Renvoi la salle courante
046     */
047    public Room getCurrentRoom(){
048        return this.aCurrentRoom;
049    }//getCurrentRoom
050
051    /**
052     * setCurrentRoom() : Permet de modifier la pièce courante.
053     */
054    public void setCurrentRoom(final Room pCurrentRoom){
055        this.aCurrentRoom = pCurrentRoom;
056    }//setCurrentRoom
057
058    /**
059     * hasLastRoom()
060     * @return Renvoi true si il y a une salle précédente
061     */
062    public boolean hasLastRoom(){
063        return !this.aLastRoom.empty();
064    }
065
066    /**
067     * getLastRoom() : Permet de récupérer la dernière Room
068     * @return Renvoi la dernière Room
069     */
070    public Room getLastRoom(){
071        return this.aLastRoom.pop();
072    }//getLastRoom
073
074    /**
075     * peekLastRoom() : Permet de récupérer la dernière Room sans la supprimer
076     * @return Renvoi la dernière Room
077     */
078    public Room peekLastRoom(){
079        return this.aLastRoom.peek();
080    }//getLastRoom
081
082
083    /**
084     * setLastRoom() : Ajoute une salle à la Stack des salles précédantes
085     * @param pLastRoom Room à ajouter à la Stack
086     */
087    public void setLastRoom(final Room pLastRoom){
088        this.aLastRoom.push(pLastRoom);
089    }//setLastRoom
090
091    /**
092     * changeRoom()
093     * @param pNextRoom nom de la Room où aller
094     */
095    public void changeRoom(final Room pNextRoom){
096//        this.setLastRoom(this.aCurrentRoom);
097        this.setCurrentRoom(pNextRoom);
098        this.aTimeLimit--;
099    }
100
101    /**
102     * getTimeLimit() : Permet de récupérer le temps restant
103     * @return Renvoi le temps restant
104     */
105    public int getTimeLimit(){
106        return this.aTimeLimit;
107    }//getLastRoom
108
109    /**
110     * getMaxWeight() : Permet de récupérer le poids max
111     * @return Le poids max
112     */
113    public int getMaxWeight(){
114        return this.aMaxWeight;
115    }//getMaxWeight
116
117    /**
118     * setMaxWeight() : Permet de régler le poids max
119     * @param pMaxWeight nouveau poids max
120     */
121    public void setMaxWeight(final int pMaxWeight){
122        this.aMaxWeight = pMaxWeight;
123    }//getMaxWeight
124//
125
126//
127    /**
128     * pickUpItem() : Permet de récupérer un item dans une Room
129     * @param pName Nom de l'item à récupérer
130     * @param pRoom Room où se trouve actuellement l'objet
131     * @return String qui dit si l'objet a été prit, est trop lourd, si l'élément n'est pas dans la salle.
132     */
133    public String pickUpItem(final String pName, final Room pRoom){
134        if(pName != null){
135            if(pRoom.getItemList().containsKey(pName)){
136                if(this.aMaxWeight >= this.aItemList.getTotalWeight()+pRoom.getItemList().getItem(pName).getWeight()){
137                    this.aItemList.setItem(pName,pRoom.getItemList().getItem(pName));
138                    pRoom.getItemList().removeItem(pName);
139                    return pName+" is now in your inventory";
140                }
141                else{
142                    return "This item is too heavy";
143                }
144            }
145            else{
146                return pName+" is not in the room";
147            }
148        }
149        else{
150            return "Get what ?";
151        }
152    }//pickUpItem
153
154    /**
155     * dropItem() : Permet de poser un item dans une Room
156     * @param pName Nom de l'item à déposer
157     * @param pRoom Room où se trouve le joueur
158     * @return String qui dit si l'objet à été déposé, ou si il n'est pas présent dans l'inventaire.
159     */
160    public String dropItem(final String pName, final Room pRoom){
161        if(pName != null){
162            if(this.aItemList.containsKey(pName)){
163                pRoom.getItemList().setItem(pName, this.aItemList.getItem(pName));
164                this.aItemList.removeItem(pName);
165                return pName+" is now in the room";
166            }
167            else{
168                return pName+" is not in your inventory";
169            }
170        }
171        else{
172            return "Remove what ?";
173        }
174    }//dropItem
175
176    /**
177     * getPlayerItem() : Permet de récupérer l'inventaire du joueur
178     * @return String renvoi l'inventaire formaté
179     */
180    public String inventory(){
181        return "Inventory : "+this.aItemList.getItemsString()+"\nTotal weight : "+this.aItemList.getTotalWeight()+" on "+this.aMaxWeight;
182    }
183
184    //PEUT-ÊTRE INUTILE
185    /**
186     * getItemList() : Permet de récupérer la liste d'item
187     * @return ItemList retourne l'itemlist du joueur
188     */
189    public ItemList getItemList(){
190        return this.aItemList;
191    }
192
193    /**
194     * getMovingCharacters() : Permet de récupérer la liste des MovingCharacters
195     * @return ItemList retourne l'itemlist des MovingCharacters
196     */
197    public ItemList getMovingCharacters(){
198        return this.aMovingCharacterList;
199    }
200
201    /**
202     * moveCharacters() : Déplace les MovingCharacters
203     */
204    public void moveCharacters(){
205      Object[] vValues = this.aMovingCharacterList.getItemList().values().toArray();
206      for(int i = 0; i < vValues.length; i++){
207          MovingCharacter vCharacter = (MovingCharacter) vValues[i];
208          vCharacter.move();
209      }
210    }
211
212     /**
213      * charge() : charge le beamer
214      * @return une String qui dit que le beamer est chargé ou qu'il n'est pas disponible
215      */
216    public String charge(final String pBeamer){
217        if(pBeamer != null){
218            if(this.aItemList.containsKey(pBeamer)){
219                if(this.aItemList.getItem(pBeamer) instanceof Beamer){
220                    return ((Beamer)this.aItemList.getItem(pBeamer)).charge(this.aCurrentRoom);
221                }
222                else{
223                    return "This is not a Beamer";
224                }
225            }
226            else{
227                return "You don't have a Beamer in your inventory";
228            }
229        }
230        else{
231            return "Charge what ?";
232        }
233    }//charge
234
235    /**
236      * fire() : décharge le beamer
237      * @return une String qui dit que le beamer est déchargé ou qu'il n'est pas chargé
238      */
239    public String fire(final String pBeamer){
240        if(pBeamer != null){
241            if(this.aItemList.containsKey(pBeamer)){
242                if(this.aItemList.getItem(pBeamer) instanceof Beamer){
243                    Beamer vBeamer = (Beamer) this.aItemList.getItem(pBeamer);
244                    if(vBeamer.isCharged()){
245                        this.changeRoom(vBeamer.fire());
246                        return "You were teleport in : "+this.getCurrentRoom().getLongDescription();
247                    }
248                    else{
249                        return "The Beamer isn't charged";
250                    }
251                }
252                else{
253                    return "This is not a Beamer";
254                }
255            }
256            else{
257                return "You don't have a Beamer in your inventory";
258            }
259        }
260        else{
261            return "Fire what ?";
262        }
263    }//charge
264}
