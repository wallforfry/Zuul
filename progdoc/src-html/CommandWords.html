001 import java.util.HashMap;
002 import java.util.Set;
003
004/**
005 * This class is part of the "World of Zuul" application.
006 * "World of Zuul" is a very simple, text based adventure game.
007 *
008 * This class holds an enumeration table of all command words known to the game.
009 * It is used to recognise commands as they are typed in.
010 *
011 * @author  Michael Kolling and David J. Barnes + D.Bureau
012 * @version 2008.03.30 + 2013.09.15
013 */
014public class CommandWords
015{
016    private HashMap <String, CommandWord> aValidCommands;
017
018    /**
019     * Constructeur par defaut
020     */
021    public CommandWords()
022    {
023        aValidCommands = new HashMap<String, CommandWord>();
024        for(CommandWord command : CommandWord.values()) {
025            if(command != CommandWord.UNKNOWN) {
026                aValidCommands.put(command.toString(), command);
027            }
028        }
029    } // CommandWords()
030
031    /**
032     * Verifie si une String fait partie des commandes valides.
033     * @param pString la String a tester
034     * true si pString est valide
035     */
036    public boolean isCommand( final String pString )
037    {
038        return this.aValidCommands.containsKey(pString);
039    } // isCommand()
040
041    /**
042     * Find the CommandWord associated with a command word.
043     * @param commandWord The word to look up.
044     * @return The CommandWord correspondng to commandWord, or UNKNOWN
045     *         if it is not a valid command word.
046     */
047    public CommandWord getCommandWord(String commandWord)
048    {
049        CommandWord command = aValidCommands.get(commandWord);
050        if(command != null) {
051            return command;
052        }
053        else {
054            return CommandWord.UNKNOWN;
055        }
056    }
057
058    /**
059     * Gets the command.
060     *
061     * @param pCommandWord the command word
062     * @return the command
063     */
064    public Command getCommand(String pCommandWord)
065    {
066        return this.getCommandWord(pCommandWord).getCommand();
067    }
068
069    /**
070     * showAll() : Permet d'extraire les mots de commande valides
071     * @return Mots de commande
072     */
073    public String showAll(){
074        String vResult = "";
075
076        for( String command : this.aValidCommands.keySet()){
077            vResult = vResult + command + ' ';
078        }
079        return vResult;
080    }//showAll
081
082} // CommandWords
