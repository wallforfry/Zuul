001import java.util.HashMap;
002import java.util.Random;
003/**
004 * Classe qui renvoi une Room au hasard
005 *
006 * @author Wallerand
007 * @version 23/05/2016
008 */
009public class RoomRandomizer
010{
011    private static String aAleaRoom;
012
013    /**
014     * Constructeur de la classe
015     */
016    public RoomRandomizer()
017    {
018    }
019
020    /**
021     * Setter de aAleaRoom
022     * @param pAleaRoom Room a utiliser comme Room al√©atoire
023     */
024    public static String setAleaRoom(final String pAleaRoom){
025        if(GameEngine.getDebug()){
026            if(pAleaRoom == null){
027                aAleaRoom = "";
028                return "Random is real";
029            }
030            else{
031                aAleaRoom = pAleaRoom;
032                return "Random set on "+aAleaRoom;
033            }
034        }
035        else{
036            return "The game isn't in Debug Mod";
037        }
038    }
039
040    /**
041     * getRandomRoom Renvoi une Room random
042     * @return A Random Room
043     */
044    public static Room getRandomRoom(final HashMap<String, Room> pRoomList){
045        if(GameEngine.getDebug() && aAleaRoom != ""){
046            return pRoomList.get(aAleaRoom);
047        }
048        else{
049            Random generator = new Random();
050            Object[] values = pRoomList.values().toArray();
051            Room randomValue = (Room) values[generator.nextInt(values.length)];
052            return randomValue;
053        }
054    }
055}
