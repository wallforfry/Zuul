001package pkg_commands;
002
003 import java.util.HashMap;
004 import java.util.Set;
005
006/**
007 * This class is part of the "World of Zuul" application.
008 * "World of Zuul" is a very simple, text based adventure game.
009 *
010 * This class holds an enumeration table of all command words known to the game.
011 * It is used to recognise commands as they are typed in.
012 *
013 * @author  Michael Kolling and David J. Barnes + D.Bureau
014 * @version 2008.03.30 + 2013.09.15
015 */
016public class CommandWords
017{
018    private HashMap <String, CommandWord> aValidCommands;
019
020    /**
021     * Constructeur par defaut
022     */
023    public CommandWords()
024    {
025        aValidCommands = new HashMap<String, CommandWord>();
026        for(CommandWord command : CommandWord.values()) {
027            if(command != CommandWord.UNKNOWN) {
028                aValidCommands.put(command.toString(), command);
029            }
030        }
031    } // CommandWords()
032
033    /**
034     * Verifie si une String fait partie des commandes valides.
035     * @param pString la String a tester
036     * true si pString est valide
037     */
038    public boolean isCommand( final String pString )
039    {
040        return this.aValidCommands.containsKey(pString);
041    } // isCommand()
042
043    /**
044     * Find the CommandWord associated with a command word.
045     * @param commandWord The word to look up.
046     * @return The CommandWord correspondng to commandWord, or UNKNOWN
047     *         if it is not a valid command word.
048     */
049    public CommandWord getCommandWord(String commandWord)
050    {
051        CommandWord command = aValidCommands.get(commandWord);
052        if(command != null) {
053            return command;
054        }
055        else {
056            return CommandWord.UNKNOWN;
057        }
058    }
059
060    /**
061     * Gets the command.
062     *
063     * @param pCommandWord the command word
064     * @return the command
065     */
066    public Command getCommand(String pCommandWord)
067    {
068        return this.getCommandWord(pCommandWord).getCommand();
069    }
070
071    /**
072     * showAll() : Permet d'extraire les mots de commande valides
073     * @return Mots de commande
074     */
075    public String showAll(){
076        String vResult = "";
077
078        for( String command : this.aValidCommands.keySet()){
079            vResult = vResult + command + ' ';
080        }
081        return vResult;
082    }//showAll
083
084} // CommandWords
