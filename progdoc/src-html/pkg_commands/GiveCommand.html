001package pkg_commands;
002
003import pkg_player.Player;
004
005import pkg_item.Character;
006
007import pkg_game.GameEngine;
008
009/**
010 * TalkCommand : Commande qui permet au player de parler avec un Character
011 */
012public class GiveCommand extends Command {
013
014        public GiveCommand() {
015        }
016
017        @Override
018        public boolean execute(Player pPlayer) {
019
020    if(this.hasSecondWord()) {
021      if(this.hasThirdWord()) {
022          if(pPlayer.getCurrentRoom().hasCharacter(this.getThirdWord())) {
023
024              Character vCharacter = (Character) pPlayer.getCurrentRoom().getCharacter(this.getThirdWord());
025
026              if(pPlayer.getItemList().containsKey(this.getSecondWord())){
027
028                  pPlayer.getItemList().setItem(vCharacter.getItem().getDescription(), vCharacter.getItem());
029                        GameEngine.aGui.println(vCharacter.getName()+" give "+vCharacter.getItem().getDescription()+" in exchange for "+this.getSecondWord());
030
031                  vCharacter.setItem(pPlayer.getItemList().getItem(this.getSecondWord()));
032                  pPlayer.getItemList().removeItem(this.getSecondWord());
033              }
034              else {
035                GameEngine.aGui.println("You don't have "+this.getSecondWord()+" in your inventory");
036              }
037
038          }
039          else {
040              GameEngine.aGui.println(this.getThirdWord()+" isn't in "+pPlayer.getCurrentRoom().getDescription());
041          }
042      }
043      else{
044        GameEngine.aGui.println("Give "+this.getSecondWord()+" to who ?");
045      }
046    }
047    else {
048      GameEngine.aGui.println("Give what ?");
049    }
050                return false;
051        }
052
053}
