001package pkg_commands;
002
003import pkg_commands.Command;
004import pkg_commands.CommandWords;
005
006import java.util.StringTokenizer;
007
008/**
009 * This class is part of the "World of Zuul" application.
010 * "World of Zuul" is a very simple, text based adventure game.
011 *
012 * This parser reads user input and tries to interpret it as an "Adventure"
013 * command. Every time it is called it reads a line from the terminal and
014 * tries to interpret the line as a two word command. It returns the command
015 * as an object of class Command.
016 *
017 * The parser has a set of known command words. It checks user input against
018 * the known commands, and if the input is not one of the known commands, it
019 * returns a command object that is marked as an unknown command.
020 *
021 * @author  Michael Kolling and David J. Barnes + D.Bureau
022 * @version 2008.03.30 + 2013.09.15
023 */
024public class Parser
025{
026    private CommandWords aValidCommands;  // (voir la classe CommandWords)
027
028    /**
029     * Constructeur par defaut qui cree les 2 objets prevus pour les attributs
030     */
031    public Parser()
032    {
033        this.aValidCommands = new CommandWords();
034    } // Parser()
035
036    /**
037     * @return The next command from the user.
038     */
039    public Command getCommand(String inputLine)
040    {
041        String vWord1 = null;
042        String vWord2 = null;
043        String vWord3 = null;
044
045        StringTokenizer tokenizer = new StringTokenizer(inputLine);
046
047        if(tokenizer.hasMoreTokens())
048            vWord1 = tokenizer.nextToken();      // get first word
049        else
050            vWord1 = null;
051        if(tokenizer.hasMoreTokens())
052            vWord2 = tokenizer.nextToken();      // get second word
053        else
054            vWord2 = null;
055        if(tokenizer.hasMoreTokens()) {
056            vWord3 = tokenizer.nextToken();
057            if(vWord3.equals("to") && tokenizer.hasMoreTokens())
058                vWord3 = tokenizer.nextToken();
059            else
060                vWord3 = null;
061        }
062        else {
063            vWord3 = null;
064        }
065
066        // note: we just ignore the rest of the input line.
067
068        // Now check whether this word is known. If so, create a command
069        // with it. If not, create a "UNKNOWN" command (for unknown command).
070        Command vCommand = aValidCommands.getCommand(vWord1);
071        if(vCommand != null) {
072                vCommand.setSecondWord(vWord2);
073          vCommand.setThirdWord(vWord3);
074        }
075        return vCommand;
076    } // getCommand()
077
078    /**
079     * showCommands() : Permet de récupérer les mots de commande déclarés dans CommandWords
080     * @return Mots de commandes
081     */
082    public String showCommands(){
083        return aValidCommands.showAll();
084    }//showCommands
085} // Parser
