001/**
002 * GoCommand : Commande permettant de changer de pi√®ce
003 */
004public class GoCommand extends Command {
005
006        public GoCommand() {
007        }
008
009        @Override
010        public boolean execute(Player pPlayer) {
011
012    String vDirection = this.getSecondWord();
013    Room vNextRoom = null;
014
015    if(vDirection != null){
016        vNextRoom = pPlayer.getCurrentRoom().getExits(vDirection);
017    }
018    else{
019        GameEngine.aGui.println("Go where ?");
020    }
021
022    if(vNextRoom == null)
023    {
024        GameEngine.aGui.println("There is no door !");
025    }//if
026    else{
027        if(vNextRoom.getExits(pPlayer.getCurrentRoom().getDescription()) != null){
028         pPlayer.setLastRoom(pPlayer.getCurrentRoom());
029        }
030        else{ //cas d'une trapdoor
031            pPlayer.setLastRoom(null);
032        }
033        pPlayer.changeRoom(vNextRoom);
034        GameEngine.aGui.println("New Current Room : "+pPlayer.getCurrentRoom().getLongDescription());
035        GameEngine.aGui.println(pPlayer.inventory());
036        GameEngine.displayImage(pPlayer);
037    }//else
038
039                return false;
040        }
041
042}
